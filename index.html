function sortByBit(arr) {
  arr.sort(function(a, b) {
    const BASE = 2;
    const REG_EXP = new RegExp("1", "g");

    let aa = a.toString(BASE);
    let bb = b.toString(BASE);
    let aaBitsCounter = (aa.match(REG_EXP) || []).length;
    let bbBitsCounter = (bb.match(REG_EXP) || []).length;

    if (aaBitsCounter > bbBitsCounter) return 1;
    if (aaBitsCounter < bbBitsCounter) return -1;
    return a - b;
  });
}
